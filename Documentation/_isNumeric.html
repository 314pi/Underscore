NAME

      _isNumeric -- Test if argument is numeric 
  

SYNOPSIS

      _isNumeric [VAR] 
  

OPTIONS

  -h      Help page 
 

DESCRIPTION

  Tests if argument is numeric. Accepted digits are '0-9,.' 
  

EXAMPLES

        CALL _IsNumeric 12 
        IF ERRORLEVEL 1 (ECHO not) ELSE ECHO:OK 
        OK 
         
        CALL _IsNumeric x12 
        IF ERRORLEVEL 1 (ECHO not) ELSE ECHO:OK 
        not 
         
        CALL _IsNumeric 1.2 
        IF ERRORLEVEL 1 (ECHO not) ELSE ECHO:OK 
        OK 
         
        CALL _IsNumeric 1a2 
        IF ERRORLEVEL 1 (ECHO not) ELSE ECHO:OK 
        not
