NAME

      getDayOfWeek -- Get day of week for a given date 
  

SYNOPSIS

      getDayOfWeek [VAR] [YYYY-MM-DD] 
  

OPTIONS

  -h      Help page 
 

DESCRIPTION

  Get day number in week: Man - Sun 0 - 6  
  

EXAMPLES

      set _dow 
      _dow=1
