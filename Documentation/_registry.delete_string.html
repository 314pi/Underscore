NAME

      _registry.delete_string -- Delete an entry from the registration database 
  

SYNOPSIS

  _registry.delete_string 'Path' 'key' 
  

OPTIONS

  -h      Help page 
  

DESCRIPTION

  _registry.delete_string deletes the entries found in the key given as first argument 
  
  Note: Patterns are case insensitive 
 

EXAMPLES

      CALL _registry.delete_string 'HKEY_LOCAL_MACHINE\SOFTWARE\ClicketyClick.dk\_registry.delete_string' 'Status' 
  
  
  
  

REQUIRES

  NOTE This function requres write access to registry (admin rights)
