NAME

      RemoveVarByType -- Remove variables by type / prefix 
  

SYNOPSIS

      RemoveVarByType [VAR] 
  

OPTIONS

  -h      Help page 
 

DESCRIPTION

  Remove all variables from the environment by the variables prefix 
  

EXAMPLES

       
              SET #x=x 
              SET #y=y 
              CALL removevarbytype #
