NAME

      spinner -- Spinner to indicate process 
  

SYNOPSIS

      spinner [type] 
  

OPTIONS

  -h      Help page 
      Sequence '-\{PIPE}/' [Default] 
  o   Sequence '_.oO'  
  

DESCRIPTION

  A textual spinner to indicate that a process is running 
  

EXAMPLES

      SET _last_line=10 
      SET _Spinner= 
      FOR /L {PCT}{PCT}x IN (1,1,{PCT}_last_line{PCT}) DO ( 
          CALL spinner 
          timeout 1 {GT}nul 
      ) 
  
 Or shorter 
      FOR /L {PCT}x IN (1,1,10) DO @CALL spinner{AMP}timeout 1 {GT}nul 
  

REQUIRES

  _BS     Set environment variable _BS to contain a backspace 
 
 

SOURCE
